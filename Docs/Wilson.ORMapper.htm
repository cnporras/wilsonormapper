Mappings.config

 Example The mappings file should have entities for each table, along with attributes for each field. Lookups allow you to map a join with a lookup table to map additional read-only lookup fields. Create views for more complicated scenarios with multiple tables, just like a table is mapped. Relationships supported are one-to-many, many-to-one (used for one-to-one), and many-to-many. Differs 

	<?xml version="1.0" encoding="utf-8" ?>
	<mappings [version="4.2"] [defaultNamespace="Namespace"]>
		<file path="file|path|resource" [embedded="true|false"] />
		<entity type="Namespace.BusinessClass" table="TableName" keyMember="MemberName[,...]"
				[keyType="Auto|Guid|User|Composite|None"] [sortOrder="FieldName ASC|DESC"]
				[changesOnly="true|false"] [readOnly="true|false"] [autoTrack="true|false"]
				[typeField="TypeDiscriminatorField"] [typeValue="DefaultValue"]
				[insertSP="InsertSP"] [updateSP="UpdateSP"] [deleteSP="DeleteSP"]>
			<attribute member="MemberName" field="FieldName" [nullValue="NullValue"] [alias="PropertyName"]
				[parameter="ParameterName"] [persistType="Persist|ReadOnly|Concurrent"] />
			<lookup member="MemberName" field="FieldName" [nullValue="NullValue"] [alias="PropertyName"]
				table="LookupTable" foreignKey="ForeignKey" lookupKey="LookupKey" />
			<relation relationship="OneToMany" [member="MemberName"] field="DestField" [alias="PropertyName"]
				type="Namespace.RelatedClass" [lazyLoad="true|false"] [cascadeDelete="true|false"]
				[filter="whereClause"] [selectSP="SelectSP"] [queryOnly="true|false"] />
			<relation relationship="ManyToOne" [member="MemberName"] field="DestField" [alias="PropertyName"]
				type="Namespace.RelatedClass" [lazyLoad="true|false"] [selectSP="SelectSP"] [queryOnly="true|false"] />
			<relation relationship="ManyToMany" [member="MemberName"] field="RelatedField" [alias="PropertyName"]
				type="Namespace.RelatedClass" [lazyLoad="true|false"] [cascadeDelete="true|false"] [queryOnly="true|false"]
				table="RelationTable" sourceField="RelationSourceField" destField="RelationDestField"
				[filter="whereClause"] [selectSP="SelectSP"] [insertSP="InsertSP"] [deleteSP="DeleteSP"] />
		</entity>
		<subEntity type="Namespace.BusinessClass" inherits="Namespace.BaseEntityClass" typeValue="Value">
			<attribute ... />
			<relation ... />
		</subEntity>
	</mappings>
